Kudingyu kdy = new Kudingyu();  //引入酷丁鱼

kdy.step(x)     //移动X步
kdy.turnLeft();// 左转
kdy.turnRight();//右转
kdy.turn(30);   //旋转30度
kdy.penup();//抬起画笔不留尾迹
kdy.pendown(); //画笔放下
kdy.setpencolor(2); //选择填充颜色用数字
kdy.fill(); //填充颜色
kdy.setpensize(1);  //设置笔的大小
kdy.setxy(100,100);//设置酷丁鱼的坐标位置
kdy.label("输入文字"); //让酷丁鱼写字
